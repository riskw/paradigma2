def pin():  
    tries = 3
    while tries >=0:
        pin = int(input('Silahkan Masukkan Pin Anda : '))
        if pin==(5000):
            print("Pin Benar, Silahkan Pilih Transaksi")
            return True
        else:
            print("Pin Salah, Silahkan Masukkan Pin Kembali")
            tries += 1
    print("Pin Salah, Kartu Terblokir")
    return False
